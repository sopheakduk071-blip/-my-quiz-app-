    <script>
        const AudioCtx = window.AudioContext || window.webkitAudioContext;
        let audioCtx;

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new AudioCtx();
            }
        }

        function playSound(type) {
            initAudio();
            if (!audioCtx) return;

            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            const now = audioCtx.currentTime;

            if (type === 'coin') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(587.33, now); // D5
                osc.frequency.setValueAtTime(880, now + 0.1);   // A5
                gain.gain.setValueAtTime(0.1, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);
                osc.start(now);
                osc.stop(now + 0.3);
            } else if (type === 'wrong') {
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(150, now);
                osc.frequency.exponentialRampToValueAtTime(80, now + 0.25);
                gain.gain.setValueAtTime(0.15, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);
                osc.start(now);
                osc.stop(now + 0.3);
            } else if (type === 'level-up') {
                const freqs = [261.63, 329.63, 392.00, 523.25];
                freqs.forEach((freq, idx) => {
                    const o = audioCtx.createOscillator();
                    const g = audioCtx.createGain();
                    o.connect(g);
                    g.connect(audioCtx.destination);
                    o.type = 'triangle';
                    o.frequency.setValueAtTime(freq, now + (idx * 0.08));
                    g.gain.setValueAtTime(0.1, now + (idx * 0.08));
                    g.gain.exponentialRampToValueAtTime(0.001, now + 0.4 + (idx * 0.08));
                    o.start(now + (idx * 0.08));
                    o.stop(now + 0.5 + (idx * 0.08));
                });
            }
        }
    </script>

                    ],

                },
                lab: {
                    inputs: [
                        { id: "v1", label: "សិស្សសរុប (នាក់)", val: 24, min: 5, max: 200, step: 1 },

                    ],
                    calc: (v1, v2) => {
                        const passed = Math.max(0, v1 - v2);
                        const pct = ((passed / v1) * 100).toFixed(1);
                        return `
                            <div><strong>ចំនួនសិស្សប្រឡងជាប់៖</strong> ${v1} - ${v2} = ${passed} នាក់</div>
                            <div class="text-xl text-yellow-300">ភាគរយប្រឡងជាប់ = (${passed} ÷ ${v1}) × 100 = <span class="text-green-300 font-bold">${pct}%</span></div>

                quiz: {

                    hint: "សិស្សជាប់មាន 18 នាក់។ គណនា (18 ÷ 24) × 100"

            }
        ];

        // Global Game State (Initialize from localStorage if available)
        let coins = parseInt(localStorage.getItem('math_quest_coins')) || 0;
        let completedQuests = JSON.parse(localStorage.getItem('math_quest_completed')) || [];
        let activeQuest = null;
        let currentTab = 'learn';

        // Update Global UI Scoreboard
        function updateScoreboard() {
            document.getElementById('score-coins').innerText = coins;
            document.getElementById('score-quests').innerText = `${completedQuests.length}/12`;
            localStorage.setItem('math_quest_coins', coins);
            localStorage.setItem('math_quest_completed', JSON.stringify(completedQuests));
        }

        // Reset Game Progress
        function resetGame() {
            if (confirm("តើប្អូនពិតជាចង់ចាប់ផ្តើមដំណើរផ្សងព្រេងឡើងវិញមែនទេ? កាក់ទាំងអស់នឹងត្រូវកំណត់មក ០ វិញ!")) {
                coins = 0;
                completedQuests = [];
                updateScoreboard();
                renderQuestList();
                closeWorkspace();
            }
        }

        // Render the 12 Quest Cards
        function renderQuestList() {
            const listContainer = document.getElementById('quest-list');
            listContainer.innerHTML = '';

            questsData.forEach((quest) => {
                const isCompleted = completedQuests.includes(quest.id);
                const isActive = activeQuest && activeQuest.id === quest.id;
                
                let cardClass = "gaming-card p-4 rounded-xl border flex items-center justify-between cursor-pointer ";
                if (isActive) {
                    cardClass += "bg-purple-950/40 border-purple-500 shadow-lg shadow-purple-500/20";
                } else if (isCompleted) {
                    cardClass += "bg-[#181d2a] border-green-500/30 hover:border-green-500/60";
                } else {
                    cardClass += "bg-[#11131e] border-purple-900/30 hover:border-purple-500/30";
                }

                const badgeHtml = isCompleted 
                    ? `<span class="text-xs bg-green-900/50 text-green-400 font-bold px-2 py-0.5 rounded-full"><i class="fa-solid fa-circle-check mr-1"></i>ឆ្លងវគ្គហើយ</span>`
                    : `<span class="text-xs bg-purple-950 text-purple-400 font-bold px-2 py-0.5 rounded-full">${quest.badge}</span>`;

                const scoreHtml = isCompleted 
                    ? `<span class="text-xs text-yellow-400 font-semibold"><i class="fa-solid fa-coins"></i> +10</span>`
                    : `<span class="text-xs text-gray-500"><i class="fa-solid fa-coins"></i> រង្វាន់ 10 កាក់</span>`;

                listContainer.innerHTML += `
                    <div class="${cardClass}" onclick="selectQuest(${quest.id})">
                        <div class="flex items-center gap-3">
                            <span class="text-3xl">${quest.icon}</span>
                            <div>
                                <div class="flex items-center gap-2">
                                    ${badgeHtml}
                                    ${scoreHtml}
                                </div>
                                <h3 class="text-sm font-bold text-white mt-1">${quest.title}</h3>
                            </div>
                        </div>
                        <div class="text-purple-400">
                            <i class="fa-solid ${isCompleted ? 'fa-circle-play text-green-400' : 'fa-chevron-right'}"></i>
                        </div>
                    </div>
                `;
            });
        }

        // Select a quest to load into Workspace
        function selectQuest(id) {
            activeQuest = questsData.find(q => q.id === id);
            
            // Switch view
            document.getElementById('welcome-panel').classList.add('hidden');
            document.getElementById('quest-workspace').classList.remove('hidden');

            // Set Header information
            document.getElementById('active-quest-icon').innerText = activeQuest.icon;
            document.getElementById('active-quest-badge').innerText = activeQuest.badge;
            document.getElementById('active-quest-title').innerText = activeQuest.title;

            // Load original question
            document.getElementById('quest-original').innerText = activeQuest.original;

            // Load pedagogic content (Learn tab)
            // 1. Info
            const infoList = document.getElementById('learn-info');
            infoList.innerHTML = '';
            activeQuest.learn.info.forEach(item => {
                infoList.innerHTML += `<div class="flex items-center gap-2"><span class="text-cyan-500">•</span> <span>${item}</span></div>`;
            });

            // 2. Tips
            document.getElementById('learn-tips').innerText = activeQuest.learn.tips;

            // 3. Steps
            const stepsList = document.getElementById('learn-steps');
            stepsList.innerHTML = '';
            activeQuest.learn.steps.forEach(step => {
                stepsList.innerHTML += `<div class="bg-[#0d0e15] p-2.5 rounded border border-purple-900/30">${step}</div>`;
            });

            // 4. Real-life
            document.getElementById('learn-reallife').innerText = activeQuest.learn.reallife;

            // Load Lab controls and computation screen
            setupLab();

            // Load Quiz
            document.getElementById('quiz-question').innerText = activeQuest.quiz.question;
            document.getElementById('quiz-unit-badge').innerText = activeQuest.unit;
            document.getElementById('quiz-answer-input').value = '';
            document.getElementById('quiz-feedback').classList.add('hidden');

            // Refresh cards to highlight current active card
            renderQuestList();
            
            // Always default to Learn tab on select
            switchTab('learn');
        }

        // Setup the Interactive Lab controls dynamically
        function setupLab() {
            const controlsContainer = document.getElementById('lab-controls');
            controlsContainer.innerHTML = '';

            activeQuest.lab.inputs.forEach((input, index) => {
                controlsContainer.innerHTML += `
                    <div class="flex flex-col gap-2">
                        <div class="flex justify-between items-center text-sm">
                            <span class="text-purple-300 font-semibold">${input.label}</span>
                            <span class="bg-[#0d0e15] px-2 py-0.5 rounded text-yellow-400 font-mono font-bold" id="lab-val-lbl-${input.id}">${input.val}</span>
                        </div>
                        <input type="range" id="lab-input-${input.id}" min="${input.min}" max="${input.max}" step="${input.step}" value="${input.val}" oninput="updateLabCalculation()" class="w-full accent-purple-500 bg-[#0d0e15] h-2 rounded-lg cursor-pointer">
                    </div>
                `;
            });

            updateLabCalculation();
        }

        // Live calculation updater for Lab
        function updateLabCalculation() {
            const inputValues = activeQuest.lab.inputs.map(input => {
                const el = document.getElementById(`lab-input-${input.id}`);
                const val = parseFloat(el.value);
                document.getElementById(`lab-val-lbl-${input.id}`).innerText = val;
                return val;
            });

            const resultHtml = activeQuest.lab.calc(...inputValues);
            document.getElementById('lab-calculation').innerHTML = resultHtml;
        }

        // Tab switcher
        function switchTab(tab) {
            currentTab = tab;
            const tabs = ['learn', 'lab', 'quiz'];
            
            tabs.forEach(t => {
                const btn = document.getElementById(`tab-${t}`);
                const content = document.getElementById(`content-${t}`);
                
                if (t === tab) {
                    btn.className = "flex-1 py-2 text-sm font-semibold rounded-lg text-white bg-purple-600 transition-all flex items-center justify-center gap-2";
                    content.classList.remove('hidden');
                } else {
                    btn.className = "flex-1 py-2 text-sm font-semibold rounded-lg text-gray-400 hover:text-white transition-all flex items-center justify-center gap-2";
                    content.classList.add('hidden');
                }
            });
        }

        // Close workspace
        function closeWorkspace() {
            activeQuest = null;
            document.getElementById('welcome-panel').classList.remove('hidden');
            document.getElementById('quest-workspace').classList.add('hidden');
            renderQuestList();
        }

        // Submit Quiz Answer
        function submitAnswer() {
            if (!activeQuest) return;

            const inputEl = document.getElementById('quiz-answer-input');
            const userAnswer = parseFloat(inputEl.value);
            const feedbackBox = document.getElementById('quiz-feedback');
            const feedbackIcon = document.getElementById('feedback-icon');
            const feedbackTitle = document.getElementById('feedback-title');
            const feedbackDesc = document.getElementById('feedback-desc');

            if (isNaN(userAnswer)) {
                feedbackBox.className = "p-4 rounded-xl flex items-center gap-3 bg-red-950/50 border border-red-500/30 text-red-400";
                feedbackIcon.innerText = "⚠️";
                feedbackTitle.innerText = "សូមបញ្ចូលលេខចម្លើយ!";
                feedbackDesc.innerText = "ប្អូនត្រូវតែគណនា រួចសរសេរចម្លើយជាលេខជាមុនសិន។";
                feedbackBox.classList.remove('hidden');
                playSound('wrong');
                return;
            }

            // Margin of error for float rounding (e.g., Quest 5 is 46.43)
            const isCorrect = Math.abs(userAnswer - activeQuest.answer) < 0.1;

            if (isCorrect) {
                feedbackBox.className = "p-4 rounded-xl flex items-center gap-3 bg-green-950/50 border border-green-500/30 text-green-400";
                feedbackIcon.innerText = "🎉";
                feedbackTitle.innerText = "ឆ្លើយបានត្រឹមត្រូវបំផុត! អស្ចារ្យណាស់!";
                
                const isNewCompletion = !completedQuests.includes(activeQuest.id);
                if (isNewCompletion) {
                    completedQuests.push(activeQuest.id);
                    coins += 10;
                    feedbackDesc.innerText = "ប្អូនទទួលបានចំណាត់ថ្នាក់ផ្កាយ ៥ និងរង្វាន់កាក់មាស 🪙 +10 កាក់!";
                    playSound('level-up');
                    triggerConfetti();
                } else {
                    feedbackDesc.innerText = "ប្អូនឆ្លើយត្រូវម្តងទៀតហើយ! ប៉ុន្តែប្អូនបានសន្សំកាក់ពីវគ្គនេះរួចហើយ។ ទៅសាកល្បងវគ្គផ្សេងទៀតណា!";
                    playSound('coin');
                }

                updateScoreboard();
                renderQuestList();
            } else {
                feedbackBox.className = "p-4 rounded-xl flex items-center gap-3 bg-red-950/50 border border-red-500/30 text-red-400";
                feedbackIcon.innerText = "❌";
                feedbackTitle.innerText = "មិនទាន់ត្រឹមត្រូវទេ សាកល្បងម្តងទៀត!";
                feedbackDesc.innerText = `គន្លឹះ៖ ${activeQuest.quiz.hint}`;
                playSound('wrong');
            }

            feedbackBox.classList.remove('hidden');
        }

        // Visual Confetti Effect on correct answer
        function triggerConfetti() {
            const container = document.getElementById('confetti-container');
            const colors = ['#8b5cf6', '#ec4899', '#f59e0b', '#10b981', '#3b82f6'];
            
            for (let i = 0; i < 60; i++) {
                const conf = document.createElement('div');
                const size = Math.random() * 8 + 6;
                const color = colors[Math.floor(Math.random() * colors.length)];
                
                conf.style.position = 'absolute';
                conf.style.width = size + 'px';
                conf.style.height = size + 'px';
                conf.style.background = color;
                conf.style.borderRadius = '50%';
                conf.style.left = Math.random() * 100 + 'vw';
                conf.style.top = '-10px';
                conf.style.opacity = Math.random();
                conf.style.transform = `rotate(${Math.random() * 360}deg)`;
                
                container.appendChild(conf);

                let y = -10;
                let x = parseFloat(conf.style.left);
                const speed = Math.random() * 5 + 3;
                const swing = Math.random() * 2 - 1;

                const interval = setInterval(() => {
                    y += speed;
                    x += swing;
                    conf.style.top = y + 'px';
                    conf.style.left = x + 'vw';
                    if (y > window.innerHeight) {
                        clearInterval(interval);
                        conf.remove();
                    }
                }, 16);
            }
        }

        // Initial setup
        window.onload = function() {
            renderQuestList();
            updateScoreboard();
        };
    </script>
</body>
</html>
